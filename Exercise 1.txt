var element1 = document.getElementsByClassName("leftanav__option__box")[0].childNodes[4];
var element2 = document.getElementsByClassName("leftanav__option__box")[0].childNodes[5];

var clonedElement2 = element2.cloneNode(true);

element2.parentNode.replaceChild(clonedElement1, element2);
element1.parentNode.replaceChild(clonedElement2, element1);
